```python
# New models needed

class Build(Base):
    id: UUID
    athlete_id: UUID
    plan_id: UUID
    goal_race_id: Optional[UUID]
    start_date: date
    end_date: date
    status: Enum[active, completed, abandoned]
    phase: str
    week: int
    
class Race(Base):
    id: UUID
    athlete_id: UUID
    build_id: Optional[UUID]
    date: date
    distance: str
    goal_time: Optional[timedelta]
    actual_time: Optional[timedelta]
    conditions: JSON
    splits: JSON
    survey_response: JSON
    analysis: JSON
    
class BuildAnalysis(Base):
    id: UUID
    build_id: UUID
    race_id: UUID
    what_worked: List[str]
    what_didnt: List[str]
    weaknesses: List[str]
    learnings_banked: List[str]
    created_at: datetime
    
class NextPhase(Base):
    id: UUID
    athlete_id: UUID
    previous_build_id: UUID
    phase_type: Enum[off_season, pre_training, build, maintenance]
    duration_weeks: int
    target_weaknesses: List[str]
    generated_plan_id: Optional[UUID]
    status: Enum[proposed, accepted, in_progress, completed]

class CalendarDay(Base):
    """Virtual model - computed from Plan + Activity join"""
    date: date
    athlete_id: UUID
    planned_workout: Optional[PlannedWorkout]
    actual_activity: Optional[Activity]
    status: Enum[future, completed, modified, missed]
    notes: List[DayNote]
    insights: List[Insight]
    
class DayNote(Base):
    id: UUID
    athlete_id: UUID
    date: date
    note_type: Enum[pre_workout, post_workout, free_text, voice_memo]
    content: JSON  # Structured for pre/post, text for free_text
    created_at: datetime
    
class CoachChat(Base):
    id: UUID
    athlete_id: UUID
    context_type: Enum[day, week, build, open]
    context_date: Optional[date]
    context_week: Optional[int]
    messages: List[ChatMessage]
    created_at: datetime
    
class ChatMessage(Base):
    role: Enum[user, coach]
    content: str
    timestamp: datetime
    context_snapshot: Optional[JSON]  # What the coach "saw" when responding
```
